**容斥原理**

设一共有 $n$ 个条件

$g(i)$ 表示 **恰好** 满足 $i$ 个条件的方案数

$f(k)$ 表示 **至少** 满足 $k$ 个条件的方案数（选 $k$ 个条件强制满足，剩下的 $n-k$ 个随意） 

$$
f(k) = \sum_{i=k}^n \binom{i}{k} g(i)
$$

**反演结论**

$$ 
g(k) = \sum_{i=k}^n (-1)^{i-k} \binom{i}{k} f(i)
$$

当我们发现 “求恰好满足 $k$ 个条件的方案数” 很困难

但 “求强制满足 $k$ 个条件（其余随意）的方案数” 相对容易时

二项式反演闪亮登场

**组合数模板**

```cpp
// 处理 C(n,m) m<n<N

const int N=20;
const int MOD=1000003;
long long fact[N],inv[N];
long long qpow(long long base,long long e){
    long long res=1;
    while(e){
        if(e&1)res=res*base%MOD;
        base=base*base%MOD;
        e>>=1;
    }
    return res;
}
void init(){
    fact[0]=fact[1]=1;
    for(int i=2;i<N;++i){
        fact[i]=i*fact[i-1]%MOD;
    }
    inv[N-1]=qpow(fact[N-1],MOD-2);
    for(int i=N-1;i;--i){
        inv[i-1]=i*inv[i]%MOD;
    }
}
long long C(int n,int m){
    if(n<0||m<0||n<m){
        return 0;
    }
    return fact[n]*inv[m]%MOD*inv[n-m]%MOD;
}
```

**二项式反演**

求恰好满足 $k$ 个条件的方案数

```cpp
vector<long long>f(n+1); // f[i]:至少满足 i 个条件的方案数

long long ans=0;

for(int i=k;i<=n;++i){
    long long term=C(i,k)*f[i]%MOD;
    if((i-k)&1){
        ans=(ans-term+MOD)%MOD;
    }
    else{
        ans=(ans+term)%MOD;
    }
}
```

```cpp
#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

const int MOD=1000003;
const int N=20;

ll fact[N],inv[N];
ll qpow(ll base,ll e){
    ll res=1;
    while(e){
        if(e&1)res=res*base%MOD;
        base=base*base%MOD;
        e>>=1;
    }
    return res;
}
void init(){
    fact[0]=fact[1]=1;
    for(int i=2;i<N;++i){
        fact[i]=i*fact[i-1]%MOD;
    }
    inv[N-1]=qpow(fact[N-1],MOD-2);
    for(int i=N-1;i;--i){
        inv[i-1]=i*inv[i]%MOD;
    }
}
ll C(int n,int m){
    if(n<0||m<0||n<m){
        return 0;
    }
    return fact[n]*inv[m]%MOD*inv[n-m]%MOD;
}

void solve(){
    int n,k;
    cin>>n>>k;
    vector<string>s(n);
    for(int i=0;i<n;++i)cin>>s[i];

    vector<ll>f(n+1);

    // 状压枚举所有子集
    for(int mask=1;mask<(1<<n);++mask){
        int y=__builtin_popcount(mask); //当前子集强制选了y个条件
        ll cnt=1; //方案数

        // 逐位检查所有字符串
        // 全是 '?'        -> 方案数 × 26
        // 只出现一种字母   -> 方案数 × 1 
        // 出现不止一种字母 -> 没有合法方案
        for(int i=0;i<s[0].size();++i){
            char tag='#';
            bool flag=0;
            for(int j=0;j<n;++j){
                if((mask>>j)&1){
                    if(s[j][i]!='?'){
                        if(tag=='#'){
                            tag=s[j][i];
                        }
                        else{
                            if(tag!=s[j][i]){
                                flag=1;
                                break;
                            }
                        }
                    }
                }
            }
            if(flag){
                cnt=0;
                break;
            }
            if(tag=='#'){
                cnt=cnt*26%MOD;
            }
            /*
            else{
                cnt=cnt*1%MOD;
            }
            */
        }

        f[y]=(f[y]+cnt)%MOD;
    }

    ll ans=0;
    for(int i=k;i<=n;++i){
        ll term=C(i,k)*f[i]%MOD;
        if((i-k)&1){
            ans=(ans-term+MOD)%MOD;
        }
        else{
            ans=(ans+term)%MOD;
        }
    }
    cout<<ans<<'\n';
}

int main(){
    cin.tie(0)->sync_with_stdio(0);
    int _=1;
    cin>>_;

    // 初始化组合数
    init();

    while(_--)solve();
}
```
